// Kode JavaScript Pencarian
function linearSearch(arr, target) {
    for (let i = 0; i < arr.length; i++) {
        if (arr[i] === target) return `Ditemukan pada indeks ${i}`;
    }
    return "Tidak ditemukan";
}

function binarySearch(arr, target) {
    let awal = 0, akhir = arr.length - 1;
    while (awal <= akhir) {
        let tengah = Math.floor((awal + akhir) / 2);
        if (arr[tengah] === target) return `Ditemukan pada indeks ${tengah}`;
        if (arr[tengah] < target) awal = tengah + 1;
        else akhir = tengah - 1;
    }
    return "Tidak ditemukan";
}

// Kode JavaScript Pengurutan
function bubbleSort(arr) {
    let n = arr.length;
    for (let i = 0; i < n-1; i++) {
        for (let j = 0; j < n-i-1; j++) {
            if (arr[j] > arr[j+1]) [arr[j], arr[j+1]] = [arr[j+1], arr[j]];
        }
    }
    return arr;
}

function mergeSort(arr) {
    if (arr.length <= 1) return arr;
    let mid = Math.floor(arr.length / 2);
    let kiri = mergeSort(arr.slice(0, mid));
    let kanan = mergeSort(arr.slice(mid));
    return merge(kiri, kanan);
}
function merge(kiri, kanan) {
    let hasil = [], i = 0, j = 0;
    while (i < kiri.length && j < kanan.length) {
        hasil.push(kiri[i] < kanan[j] ? kiri[i++] : kanan[j++]);
    }
    return hasil.concat(i < kiri.length ? kiri.slice(i) : kanan.slice(j));
}

function quickSort(arr) {
    if (arr.length <= 1) return arr;
    let pivot = arr[arr.length - 1];
    let kiri = [], kanan = [];
    for (let i = 0; i < arr.length - 1; i++) {
        arr[i] < pivot ? kiri.push(arr[i]) : kanan.push(arr[i]);
    }
    return [...quickSort(kiri), pivot, ...quickSort(kanan)];
}
